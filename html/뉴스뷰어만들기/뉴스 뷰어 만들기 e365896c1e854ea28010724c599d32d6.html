get 요청 구현 #1

  axios.get('/서버 URL')
   .then(response => {
	    data = response.data;
    })

promise를 활용한 구현 방법

javascript로 구현한 백엔드 호출 요청은 비동기(asynchronouse)로 실행된다.

  1. axios.get 메소드는 서버 URL을 GET 요청한다.
    get 메소드는 서버의 응답을 기다리지 않고 즉시 리턴한다. (몇 nano 초)
  1. then 메소드가 호출된다. 이 메소드는 콜백 함수를 호출하지 않고, 등록 하기만 한다.
    겨우 콜백 함수를 등록하는 간단한 일만 하기 때문에, then 메소드도 즉시 리턴된다. (몇 nano 초)
  1. 한참 후에 서버의 응답(response)이 웹브라우저에 도착하면, (몇 mili 초)
    아까 then 메소드가 등록한 콜백 함수가 호출되고,
    도착한 서버의 응답(response)가 이 콜백 함수의 파라미터로 전달된다.

get 요청 #2

async function test1() {
  const response = await axios.get('/서버 URL');
  const data = response.data;
}

async, await를 활용한 구현 방법
본문에 await 키워드가 있는 함수의 앞 부분에 반드시 async 키워드를 붙여야 한다.

async, await 문법을 활용하면, javascript의 비동기 작업을 동기 작업처럼 편하게 구현할 수 있다.

위 코드가 동기 작업이라면,

  1. axios.get 메소드는 즉시 리턴하지 않는다. 서버의 응답이 웹브라우저에 도착하면, 그 때 리턴한다. 이 메소드의 리턴값은, 서버의 응답이다.
  1. 그 다음 줄이 실행된다.

그러나 위 코드는 동기 작업이 아니고, 비동기 작업이다.

위와 같이 axios.get 메소드 호출이 동기(synchronous) 호출인 것처럼 편하게 구현할 수 있다.

그러나 여전히 axios.get 메소드는 비동기 실행된다.
그렇지만 마치 동기인 것처럼 위와 같이 구현해도, javascript 엔진이 잘 실행해 준다.

import { useState } from "react";

const App = () => {

        "https://newsapi.org/v2/top-headlines?country=kr&apiKey=ec72d3fbc6c54c8a8757d23b079a9430"

      console.log(e);
    }

  return (
    <div>
      <div>
        <button onClick={onClick}>불러오기</button>
      </div>
      {data && (
        <textarea
          rows={7}
          value={JSON.stringify(data, null, 2)}
          readOnly={true}
        />
      )}
    </div>
  );
}

export default App;

데이터 연동

이제는 뉴스 API를 이용해 API를 호출해보겠습니다.

useEffect를 사용해 컴포넌트가 처음 렌더링되는 시점에 API를 요청하면 됩니다.

주의할 점은 useEffect에 등록하는 함수에 async를 붙이면 안된다는 것이다.

useEffect에서 반환해야하는 값은 “뒷정리 함수”이기 때문에

useEffect 내부에서 async/await을 사용하고 싶다면,

함수 내부에 async 키워드가 붙은 또 다른 함수를 만들어 사용해야 합니다.

추가로 loading이라는 상태를 관리해 API 요청이 대기 중인지 판별하고,

요청이 대기 중일 때는 loading 값이 true 가 되고, 요청이 끝나면 loading 값이 false가 되어야 한다.

참고 : 뒷정리 함수 = 컴포넌트가 언마운트 되기 직전 또는 변경될 때 실행되는 함수

components/NewsList.js

import { useEffect, useState } from "react";
import styled from "styled-components";
import NewsItem from "./NewsItem";
import axios from "axios";

const NewsListBlock = styled.div`
  box-sizing: border-box;
  padding-bottom: 3rem;
  width: 768px;
  margin: 0 auto;
  margin-top: 2rem;
  @media screen and (max-width: 768px) {
    width: 100%;
    padding-left: 1rem;
    padding-right: 1rem;
  }
`;

const NewsList = () => {
	//useState훅을 사용하여 뉴스 목록과 로딩 상태가 저장
  const [articles, setArticles] = useState(null);
  const [loading, setLoading] = useState(false);

  useEffect(() => {
	  //async를 사용하는 함수 따로 선언
    const fetchData = async () => {
      setLoading(true); //로딩 상태를 true로 지정
      try {
	      //axios.get메서드를 호출하고 뉴스 데이터를 가져옴
        const response = await axios.get(
          "https://newsapi.org/v2/top-headlines?country=kr&apiKey=181cd9b8483e41039a15395e41f3b50b"
        );
        //가져온 뉴스 목록을 setArticles를 사용하여 컴포넌트의 상태에 저장
        setArticles(response.data.articles);
      } catch (e) { //실패시 에러 메시지 표시
        console.log(e);
      }
      //데이터를 상태에 저장하면 로딩 상태를 false로 지정
      setLoading(false);
    }
    fetchData(); //fetchData함수 호출
  }, []);

  if (loading) { //로딩 상태가 true면 NewsListBlock에 대기 중이라는 문구를 출력해서 리턴
    return <NewsListBlock>대기 중...</NewsListBlock>;
  }

  if (!articles) { //로딩 상태가 false면 null을 리턴(이를 통해 컴포넌트가 렌더링 되지 않는다)
    return null;
  }

//map()메서드를 사용하여 뉴스 목록을 반복하고, NewsItem컴포넌트를 사용해 각 뉴스 항목을 렌더링 한다. 각 뉴스 항목의 고유 식별자로 key속성이 사용된다.
  return (
    <NewsListBlock>
      {articles.map((article) => (
        <NewsItem key={article.url} article={article} />
      ))}
    </NewsListBlock>
  );
}

export default NewsList;

App에서 category 상태를 useState로 관리하겠습니다.

추가로 category 값을 업데이트하는 onSelect라는 함수도 만들어주고, category와 onSelect 함수를 Categories 컴포넌트에게 props로 전달해주겠습니다.

또한, category 값을 NewsList 컴포넌트에게도 전달해주어야 합니다.

  //onSelect는 Categories컴포넌트에서 카테고리 메뉴를 클릭하면 선택된 카테고리를 onSelect로 전달 받는다.

import { useEffect, useState } from "react";
import styled from "styled-components";
import NewsItem from "./NewsItem";
import axios from "axios";

const NewsListBlock = styled.div`
  box-sizing: border-box;
  padding-bottom: 3rem;
  width: 768px;
  margin: 0 auto;
  margin-top: 2rem;
  @media screen and (max-width: 768px) {
    width: 100%;
    padding-left: 1rem;
    padding-right: 1rem;
  }
`;

const NewsList = ({ category }) => {
  const [articles, setArticles] = useState(null);
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    const fetchData = async () => {
      setLoading(true);
      try {
        const query = category === "all" ? "" : `&category=${category}`;
        const response = await axios.get(
          `https://newsapi.org/v2/top-headlines?country=kr${query}&apiKey=ec72d3fbc6c54c8a8757d23b079a9430`
        );
        setArticles(response.data.articles);
      } catch (e) {
        console.log(e);
      }
      setLoading(false);
    }
    fetchData();
  }, [category]);

  if (loading) {
    return <NewsListBlock>대기 중...</NewsListBlock>;
  }

  if (!articles) {
    return null;
  }

  return (
    <NewsListBlock>
      {articles.map((article) => (
        <NewsItem key={article.url} article={article} />
      ))}
    </NewsListBlock>
  );
}
